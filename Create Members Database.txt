CREATE TABLE members (
NSU_ID DECIMAL(10) PRIMARY KEY,
NAME VARCHAR(150) NOT NULL,
EMAIL VARCHAR(150) NOT NULL,
DEPARTMENT VARCHAR(150),
SUBJECT VARCHAR(150) NOT NULL,
BIRTH_DATE DATE NOT NULL,
BIO VARCHAR (500),
M_PASSWORD VARCHAR (20)NOT NULL,
M_STATUS VARCHAR(150));


CREATE TABLE blogstore (
article_key VARCHAR(150) PRIMARY KEY,
NSU_ID DECIMAL(10) NOT NULL,
title VARCHAR(150) NOT NULL,
Article VARCHAR(1500) NOT NULL,
times VARCHAR(150)NOT NULL,
dates VARCHAR(150) NOT NULL);
